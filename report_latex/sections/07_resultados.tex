    \begin{insightblock}{Insights Clave}
        \begin{itemize}
            \item \highlight{tenure}, \textbf{TotalCharges} y \textbf{InternetService	extunderscore Fiber optic} aparecen entre las variables más influyentes

        \end{itemize}
